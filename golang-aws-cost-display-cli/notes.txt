awscost services
awscost services                          # all services this month + total
awscost services --month 2024-01          # all services for specific month + total
awscost services --range 2024-01 2024-06  # services per month + total per month
awscost services --top 10                 # top N services by cost this month

awscost service
awscost service ec2                       # ec2 cost this month
awscost service ec2 --month 2024-01       # ec2 cost for specific month

--name mandatory
-- month optional
--between optional

awscost service --name <name of the service>  # display data for current month
awscost service --name <name of the service> --month <year-month> #display data for that month
awscost service --name <name of the service> --between <start_year-month> <end_year-month> #display data for that range for the month




Global flags:
--region      # AWS region
--output      # table (default), json, csv, yaml
